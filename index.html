import pandas as pd import numpy as np import torch import pytorch_lightning as pl from pytorch_forecasting import TimeSeriesDataSet, TemporalFusionTransformer, QuantileLoss from pytorch_forecasting.data import GroupNormalizer from ta.momentum import RSIIndicator from ta.trend import MACD # --- 步驟 1: 資料讀取與特徵工程 --- def prepare_data(file_path): # 如果是上傳 CSV，請取消註解下一行 # df = pd.read_csv(file_path) # 範例：自動抓取數據進行演示 import yfinance as yf df = yf.download("AAPL", start="2020-01-01", end="2026-03-01") df.reset_index(inplace=True) df.columns = [c.lower() for c in df.columns] df['symbol'] = "AAPL" # 加入技術因子 (根據 2025 論文指出之顯著因子) df['rsi'] = RSIIndicator(close=df['close']).rsi() df['macd'] = MACD(close=df['close']).macd() df['returns'] = df['close'].pct_change() # TFT 必備欄位：time_idx (連續整數) df['time_idx'] = np.arange(len(df)) df = df.dropna() return df # --- 步驟 2: 建立 TFT 資料集 --- def create_dataset(df): max_prediction_length = 5 # 預測未來 5 天 max_encoder_length = 30 # 參考過去 30 天 dataset = TimeSeriesDataSet( df, time_idx="time_idx", target="close", group_ids=["symbol"], min_encoder_length=max_encoder_length // 2, max_encoder_length=max_encoder_length, min_prediction_length=1, max_prediction_length=max_prediction_length, static_categoricals=["symbol"], time_varying_known_reals=["time_idx"], time_varying_unknown_reals=["close", "volume", "rsi", "macd", "returns"], target_normalizer=GroupNormalizer(groups=["symbol"], transformation="softplus"), add_relative_time_idx=True, add_target_scales=True, add_encoder_length=True, ) return dataset # --- 步驟 3: 模型定義與建議邏輯 --- def run_prediction(): df = prepare_data("your_database.csv") dataset = create_dataset(df) dataloader = dataset.to_dataloader(train=False, batch_size=64) # 初始化 SOTA TFT 模型 tft = TemporalFusionTransformer.from_dataset( dataset, learning_rate=0.03, hidden_size=16, attention_head_size=4, dropout=0.1, loss=QuantileLoss(), # 捕捉預測的不確定性 ) # 這裡假設您已完成訓練，進行預測 # trainer = pl.Trainer(accelerator="gpu", max_epochs=10) # trainer.fit(tft, train_dataloaders=dataloader) raw_predictions = tft.predict(dataloader, mode="raw", return_x=True) # 買賣建議邏輯 (Signal Generation) # 根據 2026 策略研究：若預測價格 > 當前價格 * 1.02，則買入 last_close = df['close'].iloc[-1] pred_mean = raw_predictions.output.prediction[0, 0, 3].item() # 取第50分位數 (Median) print(f"--- 買賣建議報告 ---") print(f"目前價格: {last_close:.2f}") print(f"預測下一目標價: {pred_mean:.2f}") if pred_mean > last_close * 1.015: return "🔥 強烈建議買入 (Strong Buy)" elif pred_mean < last_close * 0.985: return "❄️ 建議賣出 (Sell)" else: return "⚖️ 觀望 (Hold)" #
